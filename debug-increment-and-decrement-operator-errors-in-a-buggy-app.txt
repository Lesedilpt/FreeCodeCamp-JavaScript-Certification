// Increment example
let counter = 10;
console.log(`Counter is currently ${counter}`);

// Use prefix increment to produce 11
let updatedCounter = ++counter;
console.log(`Counter is now ${updatedCounter}`);

// Postfix increment example
let score = 7;
console.log(`Score is currently ${score}`);

// Use postfix increment to produce 8
let finalScore = score++;
finalScore++;
console.log(`Score is still ${finalScore}`);

// Decrement example
let coins = 3;
console.log(`Coins is currently ${coins}`);

// Use prefix decrement to produce 2
let updatedCoins = --coins;
console.log(`Coins is now ${updatedCoins}`);

// Postfix decrement example
let health = 7;
console.log(`Health is currently ${health}`);

// Use postfix decrement to produce 7
let newHealth = health--;
console.log(`Health is still ${newHealth}`);
